painted_galaxy_set_precursor_flags = {
	set_global_flag = painted_galaxy_$FLAG$
	# borrowed from Precursor Selection https://steamcommunity.com/sharedfiles/filedetails/?id=3039370479
	capital_scope = {
		solar_system = {
			while = {
				count = 40
				closest_system = {
					limit = {
						NOR = {
							has_star_flag = $FLAG$
							has_star_flag = empire_home_system
						}
					}
					set_star_flag = $FLAG$
				}
			}
		}
	}
}
